# Plan 8.0 — Voice Technical Spike

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Derisk the 5 highest-uncertainty assumptions in the Voice Dictation design (`docs/superpowers/specs/2026-05-11-voice-dictation-design.md`) by building the thinnest possible vertical slice and recording empirical findings — before committing ~7000 LOC to implementation Plans 8a-8f.

**Tech Stack:** Swift 5.9+ · SwiftUI · AVAudioEngine · NSEvent (global monitor) · CGEvent · OSLog signposts · MLX-swift OR FluidAudio (one ASR engine of choice; spike gate decides)

**Spec reference:** Section 10 "Plan 8.0 acceptance gates" in the design spec.

**Branch:** `voice-spike` (worktree)

**Sequencing:** This plan must complete before Plan 8a starts. Findings file (`docs/superpowers/findings/2026-05-XX-voice-spike-findings.md`) is the handoff artifact.

### Step 2.2: Write `SpikeMicGate` to request permission and capture 3 seconds

Create `MacAllYouNeed/VoiceSpike/SpikeMicGate.swift`:

```swift
import AVFoundation
import Foundation

/// Gate 1: requests microphone permission, then captures 3 seconds of audio.
/// Returns a multi-line summary string for display in the Spike UI.
enum SpikeMicGate {
    static func run() async -> String {
        var lines: [String] = []
        let start = Date()

        // Permission
        let granted = await AVCaptureDevice.requestAccess(for: .audio)
        lines.append("AVCaptureDevice.requestAccess(.audio) → granted=\(granted)")
        guard granted else { return lines.joined(separator: "\n") }

        // Capture 3s using AVAudioEngine
        let engine = AVAudioEngine()
        let input = engine.inputNode
        let format = input.outputFormat(forBus: 0)
        lines.append("Input format: \(format.sampleRate) Hz, \(format.channelCount) ch, \(format.commonFormat.rawValue)")

        var totalFrames: AVAudioFrameCount = 0
        var peakLevel: Float = 0
        let semaphore = DispatchSemaphore(value: 0)

        input.installTap(onBus: 0, bufferSize: 4096, format: format) { buffer, _ in
            totalFrames += buffer.frameLength
            if let ch = buffer.floatChannelData?.pointee {
                for i in 0..<Int(buffer.frameLength) {
                    peakLevel = max(peakLevel, abs(ch[i]))
                }
            }
        }

        do {
            try engine.start()
        } catch {
            lines.append("engine.start() error: \(error)")
            return lines.joined(separator: "\n")
        }

        try? await Task.sleep(nanoseconds: 3_000_000_000)

        engine.stop()
        input.removeTap(onBus: 0)

        let elapsed = Date().timeIntervalSince(start)
        let durationCaptured = Double(totalFrames) / format.sampleRate
        lines.append("Captured \(totalFrames) frames = \(String(format: "%.2f", durationCaptured)) s")
        lines.append("Peak level (linear): \(String(format: "%.4f", peakLevel))")
        lines.append("Wall-clock elapsed: \(String(format: "%.2f", elapsed)) s")

        if peakLevel < 0.001 {
            lines.append("⚠ WARNING: Peak level ~0 — mic might be muted or hardware issue.")
        } else {
            lines.append("✓ Audio captured successfully.")
        }

        return lines.joined(separator: "\n")
    }
}
```

### Step 3.1: Write `SpikeHotkeyGate` with NSEvent global monitor

Create `MacAllYouNeed/VoiceSpike/SpikeHotkeyGate.swift`:

```swift
import AppKit
import Combine
import Foundation

/// Gate 2: prove whether Fn / Globe key emits press/release events
/// suitable for push-to-talk semantics.
@MainActor
final class SpikeHotkeyGate: ObservableObject {
    @Published private(set) var transcript: String = ""

    private var monitor: Any?
    private var fnDown = false
    private var startedAt: Date?

    /// Listens for ~10 seconds. User should press and release Fn multiple times.
    func run() async -> String {
        transcript = ""
        startedAt = Date()
        log("Listening for Fn key for 10s... press and release Fn 3+ times.")

        monitor = NSEvent.addGlobalMonitorForEvents(matching: .flagsChanged) { [weak self] event in
            self?.handle(event: event)
        }
        // Also install local monitor in case the app is frontmost
        let localMonitor = NSEvent.addLocalMonitorForEvents(matching: .flagsChanged) { [weak self] event in
            self?.handle(event: event)
            return event
        }

        try? await Task.sleep(nanoseconds: 10_000_000_000)

        if let monitor { NSEvent.removeMonitor(monitor) }
        NSEvent.removeMonitor(localMonitor)
        self.monitor = nil

        if !fnDown && transcript.contains("DOWN") && transcript.contains("UP") {
            return transcript + "\n✓ Fn press and release both observed."
        } else if transcript.isEmpty {
            return transcript + "\n✗ No Fn events captured. Possible reasons: external keyboard remap, Globe key remapped to language switch, Accessibility not granted."
        } else {
            return transcript + "\n⚠ Partial: only one direction observed. May need to use modifier-only hotkey lib (KeyboardShortcuts) for reliable PTT."
        }
    }

    private func handle(event: NSEvent) {
        let isFn = event.modifierFlags.contains(.function)
        let now = Date()
        let elapsed = startedAt.map { now.timeIntervalSince($0) } ?? 0
        if isFn && !fnDown {
            fnDown = true
            log(String(format: "[%.2fs] Fn DOWN (keyCode=%d)", elapsed, event.keyCode))
        } else if !isFn && fnDown {
            fnDown = false
            log(String(format: "[%.2fs] Fn UP", elapsed))
        }
    }

    private func log(_ s: String) {
        transcript += s + "\n"
    }
}
```

### Step 4.4: Write `SpikeASRGate` with real Parakeet transcription

Create `MacAllYouNeed/VoiceSpike/SpikeASRGate.swift`:

```swift
import AVFoundation
import FluidAudio
import Foundation
import OSLog

/// Gate 3: prove a local ASR model can transcribe a fixed zh+en mixed audio file.
/// MUST produce a real, non-empty transcript. No fakes.
enum SpikeASRGate {
    /// Returns multi-line summary including raw transcription, latencies, memory deltas.
    static func run() async -> String {
        var lines: [String] = []
        let signposter = VoiceSpikeLog.signposter

        // 1. Locate fixture in main app bundle (now that wav is in MacAllYouNeed/VoiceSpike/Resources/)
        guard let url = Bundle.main.url(forResource: "zh-en-mixed-5s", withExtension: "wav") else {
            return "✗ Fixture audio not found in main bundle. Did you place it under MacAllYouNeed/VoiceSpike/Resources/ and rerun `xcodegen generate`?"
        }

        // 2. Load fixture as Float32 PCM
        let loadStart = Date()
        let samples: [Float]
        do {
            samples = try Self.loadFloat32PCM(from: url)
        } catch {
            return "✗ Failed to load fixture: \(error)"
        }
        let durationSec = Double(samples.count) / 16000.0
        lines.append(String(format: "Fixture loaded: %d samples (%.2fs) in %.0f ms",
                             samples.count, durationSec, Date().timeIntervalSince(loadStart) * 1000))

        // 3. Memory before model load
        let memBefore = Self.residentMemoryMB()
        lines.append(String(format: "Resident memory before model load: %.1f MB", memBefore))

        // 4. Load Parakeet model (FluidAudio downloads on first use, cached afterward)
        let modelLoadInterval = signposter.beginInterval("modelLoad")
        let modelLoadStart = Date()
        let asrManager: AsrManager
        do {
            let models = try await AsrModels.downloadAndLoad(version: .v3)
            asrManager = AsrManager(models: models)
        } catch {
            signposter.endInterval("modelLoad", modelLoadInterval)
            return lines.joined(separator: "\n") + "\n✗ Parakeet model failed to load: \(error)\n"
                + "If FluidAudio is unavailable, run the Qwen3 stretch path (Step 4.6)."
        }
        let modelLoadMs = Date().timeIntervalSince(modelLoadStart) * 1000
        signposter.endInterval("modelLoad", modelLoadInterval)
        lines.append(String(format: "Model loaded in %.0f ms", modelLoadMs))

        let memAfter = Self.residentMemoryMB()
        lines.append(String(format: "Resident memory after model load: %.1f MB (Δ +%.1f MB)",
                             memAfter, memAfter - memBefore))

        // 5. Warmup pass (timing discarded)
        do {
            _ = try await asrManager.transcribe(samples: samples)
        } catch {
            lines.append("Warmup transcribe error: \(error)")
        }

        // 6. Three measured passes
        var passLatencies: [Double] = []
        var lastTranscript = ""
        for i in 1...3 {
            let infStart = Date()
            let interval = signposter.beginInterval("inference", "pass=\(i)")
            do {
                let result = try await asrManager.transcribe(samples: samples)
                lastTranscript = result.text
            } catch {
                lines.append("Pass \(i) ERROR: \(error)")
                signposter.endInterval("inference", interval)
                continue
            }
            signposter.endInterval("inference", interval)
            let ms = Date().timeIntervalSince(infStart) * 1000
            passLatencies.append(ms)
            lines.append(String(format: "Pass %d: %.0f ms", i, ms))
        }

        guard !passLatencies.isEmpty else {
            return lines.joined(separator: "\n") + "\n✗ All inference passes failed."
        }

        let median = passLatencies.sorted()[passLatencies.count / 2]
        lines.append(String(format: "Median inference: %.0f ms", median))
        lines.append("Last transcript: \"\(lastTranscript)\"")

        // 7. Hard pass criterion (review enforces this)
        if lastTranscript.trimmingCharacters(in: .whitespaces).isEmpty {
            lines.append("✗ FAIL: transcript is empty. Gate 3 does not pass.")
        } else {
            lines.append("✓ Real transcript produced. Now compare against ground-truth in findings.")
        }

        return lines.joined(separator: "\n")
    }

    // MARK: - helpers

    private static func loadFloat32PCM(from url: URL) throws -> [Float] {
        let file = try AVAudioFile(forReading: url)
        let format = file.processingFormat
        guard format.sampleRate == 16000, format.channelCount == 1 else {
            throw NSError(domain: "spike", code: 1, userInfo: [
                NSLocalizedDescriptionKey: "Fixture must be 16kHz mono, got \(format.sampleRate)Hz \(format.channelCount)ch"
            ])
        }
        let frames = AVAudioFrameCount(file.length)
        guard let buf = AVAudioPCMBuffer(pcmFormat: format, frameCapacity: frames) else {
            throw NSError(domain: "spike", code: 2)
        }
        try file.read(into: buf)
        guard let floatPtr = buf.floatChannelData?.pointee else { return [] }
        return Array(UnsafeBufferPointer(start: floatPtr, count: Int(buf.frameLength)))
    }

    private static func residentMemoryMB() -> Double {
        var info = mach_task_basic_info()
        var count = mach_msg_type_number_t(MemoryLayout<mach_task_basic_info>.size / MemoryLayout<integer_t>.size)
        let kr = withUnsafeMutablePointer(to: &info) {
            $0.withMemoryRebound(to: integer_t.self, capacity: Int(count)) {
                task_info(mach_task_self_, task_flavor_t(MACH_TASK_BASIC_INFO), $0, &count)
            }
        }
        guard kr == KERN_SUCCESS else { return -1 }
        return Double(info.resident_size) / (1024.0 * 1024.0)
    }
}
```

### Step 5.1: Write `SpikePasteGate`

Create `MacAllYouNeed/VoiceSpike/SpikePasteGate.swift`:

```swift
import AppKit
import ApplicationServices
import Foundation

/// Gate 4: prove that pasteboard + Cmd+V + clipboard restore works.
///
/// Two entry points:
///   • `runWithUserSwitch(text:)` — interactive, waits 5s for user to focus a target app
///     before posting Cmd+V. Used by Gate 4 button to test in TextEdit / Notes / VS Code.
///   • `runUnattended(text:)` — no human wait, only the machine work. Used by Gate 5
///     benchmark; the keystroke goes wherever focus happens to be (frontmost is the
///     spike app itself, so paste lands nowhere visible — that's expected, we only
///     measure the timing).
enum SpikePasteGate {
    static func runWithUserSwitch(text: String = "你好 from Voice spike — 测试 paste injection 工作正常") async -> String {
        var lines: [String] = []

        let trusted = AXIsProcessTrusted()
        lines.append("AXIsProcessTrusted: \(trusted)")
        if !trusted {
            lines.append("⚠ Accessibility not granted. CGEvent paste will silently fail. Trying anyway...")
        }

        let originalSnapshot = Self.snapshotPasteboard()
        lines.append("Original pasteboard had \(originalSnapshot.count) item(s)")

        lines.append("Waiting 5s — focus the target app's text field NOW...")
        try? await Task.sleep(nanoseconds: 5_000_000_000)

        let timing = Self.unattendedCore(text: text)
        lines.append(String(format: "Pasteboard set in %.0f µs", timing.setMicros))
        lines.append(String(format: "CGEvent post completed in %.0f µs", timing.postMicros))
        lines.append("✓ Cmd+V posted via CGEvent")

        try? await Task.sleep(nanoseconds: 450_000_000)
        Self.restorePasteboard(originalSnapshot)
        lines.append("✓ Original pasteboard restored")
        lines.append("👀 Verify the text appeared in the target app, and that ⌘V there now restores your previous clipboard.")

        return lines.joined(separator: "\n")
    }

    /// Pure machine work (no human wait, no restore delay). Returns measured µs.
    /// Restores pasteboard immediately. Suitable for benchmarking in Gate 5.
    @discardableResult
    static func runUnattended(text: String) -> (setMicros: Double, postMicros: Double) {
        let snapshot = Self.snapshotPasteboard()
        let timing = Self.unattendedCore(text: text)
        Self.restorePasteboard(snapshot)
        return timing
    }

    // MARK: - shared core (no human wait, no restore delay)

    private struct PasteboardSnapshot {
        let items: [[(NSPasteboard.PasteboardType, Data)]]
        var count: Int { items.count }
    }

    private static func snapshotPasteboard() -> PasteboardSnapshot {
        let pb = NSPasteboard.general
        let items = pb.pasteboardItems?.map { item -> [(NSPasteboard.PasteboardType, Data)] in
            item.types.compactMap { type in
                guard let data = item.data(forType: type) else { return nil }
                return (type, data)
            }
        } ?? []
        return PasteboardSnapshot(items: items)
    }

    private static func restorePasteboard(_ snapshot: PasteboardSnapshot) {
        let pb = NSPasteboard.general
        pb.clearContents()
        for itemTypes in snapshot.items {
            let item = NSPasteboardItem()
            for (type, data) in itemTypes {
                item.setData(data, forType: type)
            }
            pb.writeObjects([item])
        }
    }

    private static func unattendedCore(text: String) -> (setMicros: Double, postMicros: Double) {
        let pb = NSPasteboard.general

        // Measure pasteboard set
        let setStart = DispatchTime.now()
        pb.clearContents()
        pb.setString(text, forType: .string)
        let setMicros = Double(DispatchTime.now().uptimeNanoseconds - setStart.uptimeNanoseconds) / 1000.0

        // Measure CGEvent post
        let source = CGEventSource(stateID: .privateState)
        let vKey: CGKeyCode = 9 // 'v'
        let postStart = DispatchTime.now()
        if let down = CGEvent(keyboardEventSource: source, virtualKey: vKey, keyDown: true),
           let up   = CGEvent(keyboardEventSource: source, virtualKey: vKey, keyDown: false) {
            down.flags = .maskCommand
            up.flags   = .maskCommand
            down.post(tap: .cgSessionEventTap)
            up.post(tap: .cgSessionEventTap)
        }
        let postMicros = Double(DispatchTime.now().uptimeNanoseconds - postStart.uptimeNanoseconds) / 1000.0

        return (setMicros, postMicros)
    }
}
```

```swift
import AVFoundation
import FluidAudio
import Foundation
import OSLog

/// Gate 5: end-to-end machine-work timing using OSSignposter.
///
/// What we measure (real):
///   • ASR inference on a fixed audio fixture (3 passes, median)
///   • Pasteboard set + CGEvent Cmd+V post (microseconds)
///   • Pasteboard restore (microseconds)
///
/// What we explicitly do NOT measure:
///   • User speaking time (human-in-the-loop, not machine work)
///   • Sleep-simulated "capture" — real AVAudioEngine.stop() flush is ms-level and not on the bottleneck path
///   • Any 5s "switch to target app" UI wait
///
/// Plan 8a uses these numbers (not the spec's hypotheses) for UX and architecture decisions.
enum SpikeBenchmark {
    static func run() async -> String {
        let signposter = VoiceSpikeLog.signposter
        var lines: [String] = []
        lines.append("=== Gate 5: Real machine-work pipeline benchmark ===")

        // 1. Locate fixture (same as Gate 3)
        guard let url = Bundle.main.url(forResource: "zh-en-mixed-5s", withExtension: "wav") else {
            return "✗ Fixture missing. Re-run Gate 3 setup."
        }

        // 2. Load fixture (one-time cost, not on the latency path — recorded for reference)
        let loadStart = Date()
        let samples: [Float]
        do {
            samples = try Self.loadFloat32PCM(from: url)
        } catch {
            return "✗ Fixture load failed: \(error)"
        }
        lines.append(String(format: "Fixture load (one-time): %.0f ms (%d samples)",
                             Date().timeIntervalSince(loadStart) * 1000, samples.count))

        // 3. Load ASR model (one-time cost, not on the latency path)
        let modelInterval = signposter.beginInterval("modelLoad")
        let modelStart = Date()
        let asrManager: AsrManager
        do {
            let models = try await AsrModels.downloadAndLoad(version: .v3)
            asrManager = AsrManager(models: models)
        } catch {
            signposter.endInterval("modelLoad", modelInterval)
            return lines.joined(separator: "\n") + "\n✗ Model load failed: \(error)"
        }
        signposter.endInterval("modelLoad", modelInterval)
        lines.append(String(format: "Model load (one-time): %.0f ms", Date().timeIntervalSince(modelStart) * 1000))

        // 4. Warmup (timing discarded)
        _ = try? await asrManager.transcribe(samples: samples)

        // 5. Measured passes — 5 runs of (ASR + paste + restore), report median
        var totalLatencies: [Double] = []
        var asrLatencies: [Double] = []
        var pasteSetMicros: [Double] = []
        var pastePostMicros: [Double] = []
        var transcript = ""

        for i in 1...5 {
            // ASR
            let asrInterval = signposter.beginInterval("inference", "pass=\(i)")
            let asrStart = Date()
            do {
                let result = try await asrManager.transcribe(samples: samples)
                transcript = result.text
            } catch {
                lines.append("Pass \(i) ASR error: \(error)")
                signposter.endInterval("inference", asrInterval)
                continue
            }
            signposter.endInterval("inference", asrInterval)
            let asrMs = Date().timeIntervalSince(asrStart) * 1000
            asrLatencies.append(asrMs)

            // Paste (unattended — real machine work, no human wait)
            let pasteInterval = signposter.beginInterval("paste", "pass=\(i)")
            let timing = SpikePasteGate.runUnattended(text: transcript.isEmpty ? "benchmark fallback" : transcript)
            signposter.endInterval("paste", pasteInterval)
            pasteSetMicros.append(timing.setMicros)
            pastePostMicros.append(timing.postMicros)

            let totalMs = asrMs + (timing.setMicros + timing.postMicros) / 1000.0
            totalLatencies.append(totalMs)
            lines.append(String(format: "Pass %d: ASR %.0f ms · paste-set %.0f µs · paste-post %.0f µs · total %.1f ms",
                                 i, asrMs, timing.setMicros, timing.postMicros, totalMs))
        }

        guard !asrLatencies.isEmpty else {
            return lines.joined(separator: "\n") + "\n✗ All passes failed."
        }

        let asrMedian = asrLatencies.sorted()[asrLatencies.count / 2]
        let pasteSetMedian = pasteSetMicros.sorted()[pasteSetMicros.count / 2]
        let pastePostMedian = pastePostMicros.sorted()[pastePostMicros.count / 2]
        let totalMedian = totalLatencies.sorted()[totalLatencies.count / 2]

        lines.append("")
        lines.append("--- Medians (5 runs) ---")
        lines.append(String(format: "ASR inference:    %.0f ms", asrMedian))
        lines.append(String(format: "Pasteboard set:   %.0f µs", pasteSetMedian))
        lines.append(String(format: "CGEvent post:     %.0f µs", pastePostMedian))
        lines.append(String(format: "Machine-work end-to-end (ASR + paste): %.1f ms", totalMedian))
        lines.append("")
        lines.append("Last transcript: \"\(transcript)\"")
        lines.append("")
        lines.append("✓ Open Instruments → Logging → filter subsystem com.macallyouneed.spike")
        lines.append("  to see signposted intervals: modelLoad, inference (×5), paste (×5).")
        lines.append("")
        lines.append("These are the numbers Plan 8a should target. Spec § 8 hypotheses are now superseded.")

        return lines.joined(separator: "\n")
    }

    private static func loadFloat32PCM(from url: URL) throws -> [Float] {
        let file = try AVAudioFile(forReading: url)
        let format = file.processingFormat
        let frames = AVAudioFrameCount(file.length)
        guard let buf = AVAudioPCMBuffer(pcmFormat: format, frameCapacity: frames) else {
            throw NSError(domain: "spike", code: 2)
        }
        try file.read(into: buf)
        guard let floatPtr = buf.floatChannelData?.pointee else { return [] }
        return Array(UnsafeBufferPointer(start: floatPtr, count: Int(buf.frameLength)))
    }
}
```
