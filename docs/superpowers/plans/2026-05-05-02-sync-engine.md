# Plan 2: Folder-based Sync Engine

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the folder-based sync engine that watches a user-chosen directory, writes/reads encrypted envelopes (one file per record), resolves conflicts deterministically, and treats iCloud's placeholder-file behavior with care.

**Architecture:** A `SyncEngine` actor sits on top of a `FolderWatcher` protocol. Two `FolderWatcher` implementations: `FSEventsWatcher` for non-iCloud folders, `MetadataQueryWatcher` for iCloud Drive paths. The engine writes record versions as `<sync>/records/<ULID>/<deviceID>-<lamport>.envelope` so concurrent writers do not overwrite each other, and deletions as JSON tombstones in `<sync>/tombstones/<ULID>.<kind>.tomb`. A `Manifest` JSON file holds device list, KDF params, schema version, and current key version. Conflict resolution: Lamport-first, then `modified`, then `deviceID` (lexicographic). All envelope encryption uses Plan 1's `Cipher` with the sync key (Plan 1's `KeyManager.deriveSyncKey`).

**Tech Stack:** CoreServices (FSEvents), Foundation (`NSMetadataQuery`), CryptoKit (via Plan 1's `Cipher`), Swift concurrency (actors).

**Reads from spec:** §3 (decisions 5, 6, 14), §5 (Folder-based sync, Envelope format, Categories synced, Conflict policy, File watching, iCloud throttled hydration).

**Produces working software:** an integration test spins up two `SyncEngine` instances pointing at a shared temp folder; appending a record on engine A surfaces it on engine B; deleting on B propagates back to A; a forced lamport-tied conflict is resolved deterministically.

```swift
import Foundation

public struct SyncPaths {
    public let root: URL

    public init(root: URL) { self.root = root }

    public var records: URL { root.appendingPathComponent("records", isDirectory: true) }
    public var tombstones: URL { root.appendingPathComponent("tombstones", isDirectory: true) }
    public var manifest: URL { root.appendingPathComponent("manifest.json") }

    public func recordDirectory(for id: RecordID) -> URL {
        records.appendingPathComponent(id.rawValue, isDirectory: true)
    }

    public func envelopeURL(for metadata: EnvelopeMetadata) -> URL {
        recordDirectory(for: metadata.id)
            .appendingPathComponent("\(metadata.deviceID.rawValue)-\(metadata.lamport).envelope")
    }

    public func envelopeURLs(for id: RecordID) throws -> [URL] {
        let dir = recordDirectory(for: id)
        guard FileManager.default.fileExists(atPath: dir.path) else { return [] }
        return try FileManager.default.contentsOfDirectory(at: dir, includingPropertiesForKeys: nil)
            .filter { $0.pathExtension == "envelope" }
    }

    public func tombstoneURL(for id: RecordID, kind: RecordKind) -> URL {
        tombstones.appendingPathComponent("\(id.rawValue).\(kind.rawValue).tomb")
    }

    public func createIfNeeded() throws {
        try FileManager.default.createDirectory(at: records, withIntermediateDirectories: true)
        try FileManager.default.createDirectory(at: tombstones, withIntermediateDirectories: true)
    }
}
```

```swift
import Foundation
import Security

public struct Manifest: Codable, Equatable {
    public struct Device: Codable, Equatable {
        public let id: DeviceID
        public var name: String
        public var lastSeen: Date
        public init(id: DeviceID, name: String, lastSeen: Date) {
            self.id = id; self.name = name; self.lastSeen = lastSeen
        }
    }

    public var schemaVersion: Int
    public var keyVersion: KeyVersion
    public var kdfParams: KDFParameters
    public var kdfSalt: Data
    public var devices: [Device]

    public init(schemaVersion: Int, keyVersion: KeyVersion, kdfParams: KDFParameters,
                kdfSalt: Data, devices: [Device]) {
        self.schemaVersion = schemaVersion
        self.keyVersion = keyVersion
        self.kdfParams = kdfParams
        self.kdfSalt = kdfSalt
        self.devices = devices
    }

    public static func bootstrap(deviceID: DeviceID, deviceName: String) -> Manifest {
        var salt = Data(count: 16)
        _ = salt.withUnsafeMutableBytes { SecRandomCopyBytes(kSecRandomDefault, $0.count, $0.baseAddress!) }
        return Manifest(
            schemaVersion: 1,
            keyVersion: .v1,
            kdfParams: .defaultV1,
            kdfSalt: salt,
            devices: [Device(id: deviceID, name: deviceName, lastSeen: Date())]
        )
    }

    public static func encode(_ m: Manifest) throws -> Data {
        let enc = JSONEncoder()
        enc.outputFormatting = [.prettyPrinted, .sortedKeys]
        enc.dateEncodingStrategy = .iso8601
        return try enc.encode(m)
    }

    public static func decode(_ data: Data) throws -> Manifest {
        let dec = JSONDecoder()
        dec.dateDecodingStrategy = .iso8601
        return try dec.decode(Manifest.self, from: data)
    }

    public func write(to url: URL) throws {
        try Self.encode(self).write(to: url, options: .atomic)
    }

    public static func read(from url: URL) throws -> Manifest {
        try Self.decode(try Data(contentsOf: url))
    }
}
```

```swift
import Foundation
import CryptoKit

public enum EnvelopeIOError: Error {
    case decodeFailed
}

public enum EnvelopeIO {
    private struct Wire: Codable {
        let metadata: EnvelopeMetadata
        let body: Data
    }

    public static func write(metadata: EnvelopeMetadata, body: Data, to url: URL, key: SymmetricKey) throws {
        let wire = Wire(metadata: metadata, body: body)
        let plaintext = try JSONEncoder().encode(wire)
        let envelope = try Cipher.seal(plaintext, with: key)
        try FileManager.default.createDirectory(at: url.deletingLastPathComponent(), withIntermediateDirectories: true)
        try envelope.combined.write(to: url, options: .atomic)
    }

    public static func read(from url: URL, key: SymmetricKey) throws -> SyncRecord {
        let raw = try Data(contentsOf: url)
        let plaintext = try Cipher.open(Envelope(combined: raw), with: key)
        let wire = try JSONDecoder().decode(Wire.self, from: plaintext)
        return SyncRecord(metadata: wire.metadata, body: wire.body)
    }

    public static func writeTombstone(_ tombstone: SyncTombstone, to url: URL) throws {
        let enc = JSONEncoder()
        enc.dateEncodingStrategy = .iso8601
        enc.outputFormatting = [.sortedKeys]
        try FileManager.default.createDirectory(at: url.deletingLastPathComponent(), withIntermediateDirectories: true)
        try enc.encode(tombstone).write(to: url, options: .atomic)
    }

    public static func readTombstone(from url: URL) throws -> SyncTombstone {
        let dec = JSONDecoder()
        dec.dateDecodingStrategy = .iso8601
        return try dec.decode(SyncTombstone.self, from: try Data(contentsOf: url))
    }
}
```

```swift
import Foundation

public struct ConflictResolution {
    public let winner: SyncRecord
    public let losers: [SyncRecord]
}

public enum ConflictResolver {
    /// Last-write-wins by Lamport, then `modified`, then `deviceID` (lexicographic, larger wins).
    public static func resolve(_ records: [SyncRecord]) -> ConflictResolution {
        let sorted = records.sorted { a, b in
            if a.metadata.lamport != b.metadata.lamport {
                return a.metadata.lamport < b.metadata.lamport
            }
            if a.metadata.modified != b.metadata.modified {
                return a.metadata.modified < b.metadata.modified
            }
            return a.metadata.deviceID.rawValue < b.metadata.deviceID.rawValue
        }
        let winner = sorted.last!
        let losers = Array(sorted.dropLast())
        return ConflictResolution(winner: winner, losers: losers)
    }
}
```

```swift
import Foundation
import CoreServices

public final class FSEventsWatcher: FolderWatcher {
    private let root: URL
    private var stream: FSEventStreamRef?
    private var callback: (([String]) -> Void)?

    public init(root: URL) { self.root = root }

    public func start(callback: @escaping ([String]) -> Void) throws {
        self.callback = callback
        let context = UnsafeMutablePointer<FSEventStreamContext>.allocate(capacity: 1)
        context.initialize(to: FSEventStreamContext(
            version: 0,
            info: Unmanaged.passUnretained(self).toOpaque(),
            retain: nil, release: nil, copyDescription: nil
        ))
        let cb: FSEventStreamCallback = { _, info, count, eventPaths, _, _ in
            guard let info else { return }
            let watcher = Unmanaged<FSEventsWatcher>.fromOpaque(info).takeUnretainedValue()
            let paths = unsafeBitCast(eventPaths, to: NSArray.self) as? [String] ?? []
            watcher.callback?(paths)
        }
        let pathsToWatch: CFArray = [root.path] as CFArray
        let stream = FSEventStreamCreate(
            kCFAllocatorDefault,
            cb,
            context,
            pathsToWatch,
            FSEventStreamEventId(kFSEventStreamEventIdSinceNow),
            0.5,        // latency seconds
            FSEventStreamCreateFlags(
                kFSEventStreamCreateFlagFileEvents |
                kFSEventStreamCreateFlagNoDefer |
                kFSEventStreamCreateFlagUseCFTypes
            )
        )
        context.deallocate()
        guard let stream else { throw NSError(domain: "FSEventsWatcher", code: -1) }
        self.stream = stream
        FSEventStreamSetDispatchQueue(stream, DispatchQueue(label: "fseventswatcher.\(root.lastPathComponent)"))
        FSEventStreamStart(stream)
    }

    public func stop() {
        guard let stream else { return }
        FSEventStreamStop(stream)
        FSEventStreamInvalidate(stream)
        FSEventStreamRelease(stream)
        self.stream = nil
        self.callback = nil
    }
}
```

```swift
import Foundation

/// Watches an iCloud Drive folder using `NSMetadataQuery`. Use this when the
/// sync root is inside `~/Library/Mobile Documents`.
///
/// Note: end-to-end iCloud testing requires a real iCloud account and a long
/// signing/entitlement loop. Plan 2 ships this implementation but exercises
/// it only in manual QA. Plan 0/Plan 7 cover the entitlements; the integration
/// test uses `FSEventsWatcher` against a temp directory.
///
/// This watcher only reports changed paths. It must not call
/// `startDownloadingUbiquitousItem` directly; `SyncEngine` routes iCloud
/// placeholder hydration through `HydrationQueue` so concurrency, priority,
/// and retry behavior stay centralized.
public final class MetadataQueryWatcher: FolderWatcher {
    private let root: URL
    private let query: NSMetadataQuery
    private var observers: [NSObjectProtocol] = []
    private var callback: (([String]) -> Void)?

    public init(root: URL) {
        self.root = root
        self.query = NSMetadataQuery()
    }

    public func start(callback: @escaping ([String]) -> Void) throws {
        self.callback = callback
        query.searchScopes = [root]
        query.predicate = NSPredicate(format: "%K LIKE '*'", NSMetadataItemFSNameKey)
        query.notificationBatchingInterval = 0.5
        query.operationQueue = OperationQueue()

        let center = NotificationCenter.default
        let names: [Notification.Name] = [
            .NSMetadataQueryDidFinishGathering,
            .NSMetadataQueryDidUpdate,
        ]
        for name in names {
            let token = center.addObserver(forName: name, object: query, queue: query.operationQueue) { [weak self] note in
                self?.handle(note: note)
            }
            observers.append(token)
        }
        DispatchQueue.main.async { _ = self.query.start() }
    }

    public func stop() {
        query.stop()
        observers.forEach(NotificationCenter.default.removeObserver(_:))
        observers.removeAll()
        callback = nil
    }

    private func handle(note: Notification) {
        let items = (query.results as? [NSMetadataItem]) ?? []
        var paths: [String] = []
        for item in items {
            if let url = item.value(forAttribute: NSMetadataItemURLKey) as? URL {
                paths.append(url.path)
            }
        }
        if !paths.isEmpty { callback?(paths) }
    }
}
```

```swift
import Foundation

public actor HydrationQueue {
    public enum Priority: Int { case normal = 0, high = 1 }

    private struct Job { let url: URL; let priority: Priority }

    private var jobs: [Job] = []
    private var inflight = 0
    private let maxConcurrent: Int
    private var autoStart: Bool
    private let runner: (URL) async throws -> Void
    private var continuations: [CheckedContinuation<Void, Never>] = []

    public init(maxConcurrent: Int, autoStart: Bool = true, runner: @escaping (URL) async throws -> Void) {
        self.maxConcurrent = maxConcurrent
        self.autoStart = autoStart
        self.runner = runner
    }

    public func enqueue(url: URL, priority: Priority) {
        jobs.append(Job(url: url, priority: priority))
        jobs.sort { $0.priority.rawValue > $1.priority.rawValue }
        if autoStart {
            Task { await self.tryDispatch() }
        }
    }

    public func start() async {
        autoStart = true
        await tryDispatch()
    }

    private func tryDispatch() async {
        while inflight < maxConcurrent, let job = jobs.first {
            jobs.removeFirst()
            inflight += 1
            Task { [job] in
                await self.run(job: job, attempt: 0)
            }
        }
    }

    private func run(job: Job, attempt: Int) async {
        do {
            try await runner(job.url)
        } catch {
            if attempt < 3 {
                let delay = UInt64(pow(2.0, Double(attempt)) * 200_000_000) // 200ms * 2^n
                try? await Task.sleep(nanoseconds: delay)
                await run(job: job, attempt: attempt + 1)
                return
            }
        }
        await self.complete()
    }

    private func complete() async {
        inflight -= 1
        if jobs.isEmpty && inflight == 0 {
            continuations.forEach { $0.resume() }
            continuations.removeAll()
        }
        await tryDispatch()
    }

    public func drain() async {
        if jobs.isEmpty && inflight == 0 { return }
        await withCheckedContinuation { cont in
            continuations.append(cont)
        }
    }
}
```

```swift
import Foundation

public enum TombstoneJanitor {
    public static func sweep(paths: SyncPaths, maxAgeDays: Int) {
        let log = Logging.logger(for: "sync", category: "tombstone-janitor")
        let cutoff = Date().addingTimeInterval(-Double(maxAgeDays * 86_400))
        guard let entries = try? FileManager.default.contentsOfDirectory(at: paths.tombstones,
                                                                         includingPropertiesForKeys: [.contentModificationDateKey]) else { return }
        for url in entries where url.pathExtension == "tomb" {
            if let mod = (try? url.resourceValues(forKeys: [.contentModificationDateKey]).contentModificationDate),
               mod < cutoff {
                try? FileManager.default.removeItem(at: url)
                log.debug("Swept tombstone \(url.lastPathComponent, privacy: .public)")
            }
        }
    }
}
```

```swift
import Foundation
import CryptoKit

public actor SyncEngine {
    public typealias Subscriber = @Sendable (SyncEvent) -> Void

    private let paths: SyncPaths
    private let key: SymmetricKey
    private let deviceID: DeviceID
    private let watcher: FolderWatcher
    private var subscribers: [UUID: Subscriber] = [:]
    private var seenIDs: Set<String> = []
    private let log = Logging.logger(for: "sync", category: "engine")

    public init(paths: SyncPaths, syncKey: SymmetricKey, deviceID: DeviceID, watcher: FolderWatcher) {
        self.paths = paths
        self.key = syncKey
        self.deviceID = deviceID
        self.watcher = watcher
    }

    public func start() async throws {
        try paths.createIfNeeded()
        // Initial scan
        await scanAll()
        try watcher.start { [weak self] paths in
            Task { await self?.handleChanged(paths: paths) }
        }
    }

    public func stop() {
        watcher.stop()
        subscribers.removeAll()
    }

    @discardableResult
    public func subscribe(_ subscriber: @escaping Subscriber) -> UUID {
        let id = UUID()
        subscribers[id] = subscriber
        return id
    }

    public func unsubscribe(_ id: UUID) {
        subscribers.removeValue(forKey: id)
    }

    public func put(metadata: EnvelopeMetadata, body: Data) async throws {
        let url = paths.envelopeURL(for: metadata.id)
        try EnvelopeIO.write(metadata: metadata, body: body, to: url, key: key)
        seenIDs.insert(metadata.id.rawValue)
    }

    public func delete(id: RecordID) async throws {
        let url = paths.tombstoneURL(for: id)
        try Data().write(to: url, options: .atomic)
        // Remove the envelope if present
        try? FileManager.default.removeItem(at: paths.envelopeURL(for: id))
    }

    private func scanAll() async {
        let envURLs = (try? FileManager.default.contentsOfDirectory(at: paths.records, includingPropertiesForKeys: nil)) ?? []
        let recordsByID = Dictionary(grouping: envURLs.filter { $0.pathExtension == "envelope" }) {
            $0.deletingPathExtension().lastPathComponent
        }
        for (idString, urls) in recordsByID {
            await ingest(idString: idString, urls: urls)
        }
        let tombs = (try? FileManager.default.contentsOfDirectory(at: paths.tombstones, includingPropertiesForKeys: nil)) ?? []
        for url in tombs where url.pathExtension == "tomb" {
            let idString = url.deletingPathExtension().lastPathComponent
            if let id = RecordID(rawValue: idString) {
                emit(.recordDeleted(id, .clipboardItem)) // kind placeholder; consumers may filter
            }
        }
    }

    private func handleChanged(paths changed: [String]) async {
        let interestingIDs = changed.compactMap { path -> String? in
            let url = URL(fileURLWithPath: path)
            if url.pathExtension == "envelope" || url.pathExtension == "tomb" {
                return url.deletingPathExtension().lastPathComponent
            }
            return nil
        }
        for idString in Set(interestingIDs) {
            let env = paths.records.appendingPathComponent("\(idString).envelope")
            let tomb = self.paths.tombstoneURL(for: RecordID(rawValue: idString) ?? .generate())
            if FileManager.default.fileExists(atPath: tomb.path) {
                if let id = RecordID(rawValue: idString) {
                    emit(.recordDeleted(id, .clipboardItem))
                }
                continue
            }
            if FileManager.default.fileExists(atPath: env.path) {
                await ingest(idString: idString, urls: [env])
            }
        }
        _ = env  // silence unused if linter complains in future revisions
    }

    private func ingest(idString: String, urls: [URL]) async {
        var records: [SyncRecord] = []
        for url in urls {
            do {
                records.append(try EnvelopeIO.read(from: url, key: key))
            } catch {
                log.error("Failed to read envelope \(url.lastPathComponent): \(error.localizedDescription)")
            }
        }
        if records.isEmpty { return }
        if records.count == 1 {
            emit(.recordReceived(records[0]))
        } else {
            let res = ConflictResolver.resolve(records)
            emit(.conflict(res))
            emit(.recordReceived(res.winner))
        }
    }

    private func emit(_ event: SyncEvent) {
        for (_, sub) in subscribers { sub(event) }
    }
}
```

> Note: the unused-variable comment at the end of `handleChanged` is intentional — earlier drafts referenced `env` outside the loop. Remove if the compiler complains; left here as a guard against accidental scope changes during refactor.

```swift
import XCTest
import CryptoKit
@testable import Core

final class SyncEngineIntegrationTests: XCTestCase {
    func testRecordPropagatesAcrossEngines() async throws {
        let dir = FileManager.default.temporaryDirectory.appendingPathComponent("sei-\(UUID())", isDirectory: true)
        try FileManager.default.createDirectory(at: dir, withIntermediateDirectories: true)
        defer { try? FileManager.default.removeItem(at: dir) }

        let paths = SyncPaths(root: dir)
        try paths.createIfNeeded()
        let key = SymmetricKey(size: .bits256)

        let aDevice = DeviceID.generate()
        let bDevice = DeviceID.generate()
        let aWatcher = FSEventsWatcher(root: dir)
        let bWatcher = FSEventsWatcher(root: dir)

        let engineA = SyncEngine(paths: paths, syncKey: key, deviceID: aDevice, watcher: aWatcher)
        let engineB = SyncEngine(paths: paths, syncKey: key, deviceID: bDevice, watcher: bWatcher)

        let received = expectation(description: "B received A's record")
        await engineB.subscribe { event in
            if case let .recordReceived(record) = event,
               record.metadata.deviceID == aDevice {
                received.fulfill()
            }
        }

        try await engineA.start()
        try await engineB.start()
        defer { Task { await engineA.stop(); await engineB.stop() } }

        // Give watchers a moment to register
        try await Task.sleep(nanoseconds: 300_000_000)

        let id = RecordID.generate()
        let now = Date()
        let meta = EnvelopeMetadata(kind: .clipboardItem, id: id, created: now, modified: now,
                                    deviceID: aDevice, lamport: 1)
        try await engineA.put(metadata: meta, body: try JSONEncoder().encode(["text": "hi"]))

        await fulfillment(of: [received], timeout: 5.0)
    }

    func testDeletionPropagates() async throws {
        let dir = FileManager.default.temporaryDirectory.appendingPathComponent("sed-\(UUID())", isDirectory: true)
        try FileManager.default.createDirectory(at: dir, withIntermediateDirectories: true)
        defer { try? FileManager.default.removeItem(at: dir) }

        let paths = SyncPaths(root: dir); try paths.createIfNeeded()
        let key = SymmetricKey(size: .bits256)

        let id = RecordID.generate()
        let meta = EnvelopeMetadata(kind: .clipboardItem, id: id, created: Date(), modified: Date(),
                                    deviceID: DeviceID.generate(), lamport: 1)
        try EnvelopeIO.write(metadata: meta, body: Data(), to: paths.envelopeURL(for: id), key: key)

        let watcher = FSEventsWatcher(root: dir)
        let engine = SyncEngine(paths: paths, syncKey: key, deviceID: DeviceID.generate(), watcher: watcher)

        let deleted = expectation(description: "engine reported deletion")
        await engine.subscribe { event in
            if case let .recordDeleted(rid, _) = event, rid == id { deleted.fulfill() }
        }
        try await engine.start()
        defer { Task { await engine.stop() } }

        try await Task.sleep(nanoseconds: 300_000_000)
        try await engine.delete(id: id)

        await fulfillment(of: [deleted], timeout: 5.0)
    }
}
```
