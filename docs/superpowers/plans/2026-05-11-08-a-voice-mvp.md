# Voice MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first production Voice Dictation path: a configurable voice shortcut records audio, Qwen3-ASR transcribes zh/en speech, a Mini HUD shows state, text is pasted into the focused app, and the result is stored.

**Architecture:** Keep production code under `MacAllYouNeed/Voice/` for AppKit/FluidAudio runtime pieces and `Shared/Sources/Core/Voice/` for storage models, protocols, and pure logic. Plan 8a is the single owner of all voice-related `clipboard.sqlite` schema changes required by later plans. Use the measured Plan 8.0 constraints: signed hardened-runtime audio entitlement is required, Qwen3 f32 is accurate but batch-oriented, Fn / Globe is physically capturable but can conflict with input-source switching, and CGEvent paste is viable when Accessibility is granted.

**Implementation amendment (2026-05-11):** Default activation is `⌃⌥Space` in Toggle mode. Hold-to-talk remains available as a user setting. Sections below that name `FnPTTMonitor` are superseded by `VoiceActivationMonitor`, `VoiceActivationSettings`, and `VoiceShortcutMatcher`.

**Tech Stack:** Swift 5.9 · SwiftUI/AppKit · AVAudioEngine · FluidAudio Qwen3-ASR · GRDB · CGEvent · NSEvent flagsChanged · OSLog signposts

```swift
import Foundation
import GRDB

public final class VoiceTranscriptStore: @unchecked Sendable {
    private let db: Database

    public init(database: Database) {
        self.db = database
    }

    @discardableResult
    public func save(_ draft: VoiceTranscriptDraft) throws -> VoiceTranscript {
        let id = UUID().uuidString
        let durationMs = max(0, Int((draft.endedAt.timeIntervalSince(draft.startedAt) * 1000).rounded()))
        try db.queue.write { conn in
            try conn.execute(sql: """
                INSERT INTO voice_transcripts (
                    id, started_at, ended_at, duration_ms, raw_text, cleaned_text,
                    app_bundle_id, language, model_identifier, audio_path
                )
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            """, arguments: [
                id,
                Self.millis(draft.startedAt),
                Self.millis(draft.endedAt),
                durationMs,
                draft.rawText,
                draft.cleanedText,
                draft.appBundleID,
                draft.language.rawValue,
                draft.modelIdentifier,
                draft.audioPath
            ])
        }
        return VoiceTranscript(
            id: id,
            startedAt: draft.startedAt,
            endedAt: draft.endedAt,
            durationMs: durationMs,
            rawText: draft.rawText,
            cleanedText: draft.cleanedText,
            appBundleID: draft.appBundleID,
            language: draft.language,
            modelIdentifier: draft.modelIdentifier,
            audioPath: draft.audioPath
        )
    }

    public func fetch(id: String) throws -> VoiceTranscript? {
        try db.queue.read { conn in
            try Row.fetchOne(conn, sql: """
                SELECT id, started_at, ended_at, duration_ms, raw_text, cleaned_text,
                       app_bundle_id, language, model_identifier, audio_path
                FROM voice_transcripts WHERE id = ?
            """, arguments: [id]).map(Self.transcript(from:))
        }
    }

    private static func transcript(from row: Row) -> VoiceTranscript {
        VoiceTranscript(
            id: row["id"],
            startedAt: Date(timeIntervalSince1970: Double(row["started_at"] as Int64) / 1000),
            endedAt: Date(timeIntervalSince1970: Double(row["ended_at"] as Int64) / 1000),
            durationMs: Int(row["duration_ms"] as Int64),
            rawText: row["raw_text"],
            cleanedText: row["cleaned_text"],
            appBundleID: row["app_bundle_id"],
            language: VoiceLanguage(rawValue: row["language"]) ?? .unknown,
            modelIdentifier: row["model_identifier"],
            audioPath: row["audio_path"]
        )
    }

    private static func millis(_ date: Date) -> Int64 {
        Int64((date.timeIntervalSince1970 * 1000).rounded())
    }
}
```

```swift
import AVFoundation
import Foundation
import Observation

struct CapturedAudio: Sendable {
    let samples: [Float]
    let sampleRate: Double
    let startedAt: Date
    let endedAt: Date
    let peakLevel: Float
}

@MainActor
@Observable
final class AudioCaptureService {
    private var engine: AVAudioEngine?
    private var samples: [Float] = []
    private var startedAt: Date?
    private(set) var peakLevel: Float = 0

    func requestPermission() async -> Bool {
        await AVCaptureDevice.requestAccess(for: .audio)
    }

    func start() throws {
        samples.removeAll(keepingCapacity: true)
        peakLevel = 0
        startedAt = Date()

        let engine = AVAudioEngine()
        let input = engine.inputNode
        let format = input.outputFormat(forBus: 0)
        input.installTap(onBus: 0, bufferSize: 4096, format: format) { [weak self] buffer, _ in
            let mono = Self.floatMonoSamples(from: buffer)
            Task { @MainActor in
                self?.samples.append(contentsOf: mono)
                self?.peakLevel = max(self?.peakLevel ?? 0, mono.map(abs).max() ?? 0)
            }
        }
        try engine.start()
        self.engine = engine
    }

    func stop() -> CapturedAudio? {
        guard let engine, let startedAt else { return nil }
        let endedAt = Date()
        let sampleRate = engine.inputNode.outputFormat(forBus: 0).sampleRate
        engine.inputNode.removeTap(onBus: 0)
        engine.stop()
        self.engine = nil
        self.startedAt = nil
        return CapturedAudio(
            samples: samples,
            sampleRate: sampleRate,
            startedAt: startedAt,
            endedAt: endedAt,
            peakLevel: peakLevel
        )
    }

    static func floatMonoSamples(from buffer: AVAudioPCMBuffer) -> [Float] {
        guard let channels = buffer.floatChannelData else { return [] }
        let frameCount = Int(buffer.frameLength)
        let channelCount = Int(buffer.format.channelCount)
        guard channelCount > 0 else { return [] }
        return (0 ..< frameCount).map { frame in
            var sum: Float = 0
            for channel in 0 ..< channelCount {
                sum += channels[channel][frame]
            }
            return sum / Float(channelCount)
        }
    }
}
```

```swift
        guard #available(macOS 15, *) else { throw Qwen3EngineError.unsupportedOS }

        try await next.loadModels(from: cacheDir)
        manager = next
        return next
    }
```

```swift
import AppKit
import Foundation

@MainActor
final class FnPTTMonitor {
    private var globalMonitor: Any?
    private var localMonitor: Any?
    private var isDown = false
    private var staleTask: Task<Void, Never>?

    var onDown: (() -> Void)?
    var onUp: (() -> Void)?

    func start() {
        stop()
        globalMonitor = NSEvent.addGlobalMonitorForEvents(matching: .flagsChanged) { [weak self] event in
            Task { @MainActor in self?.handle(event) }
        }
        localMonitor = NSEvent.addLocalMonitorForEvents(matching: .flagsChanged) { [weak self] event in
            Task { @MainActor in self?.handle(event) }
            return event
        }
    }

    func stop() {
        if let globalMonitor { NSEvent.removeMonitor(globalMonitor) }
        if let localMonitor { NSEvent.removeMonitor(localMonitor) }
        globalMonitor = nil
        localMonitor = nil
        staleTask?.cancel()
        staleTask = nil
        if isDown {
            isDown = false
            onUp?()
        }
    }

    private func handle(_ event: NSEvent) {
        let nextDown = event.modifierFlags.contains(.function)
        if nextDown, !isDown {
            isDown = true
            onDown?()
            scheduleStaleRelease()
        } else if !nextDown, isDown {
            isDown = false
            staleTask?.cancel()
            staleTask = nil
            onUp?()
        }
    }

    private func scheduleStaleRelease() {
        staleTask?.cancel()
        staleTask = Task { @MainActor in
            try? await Task.sleep(for: .seconds(30))
            guard !Task.isCancelled, isDown else { return }
            isDown = false
            onUp?()
        }
    }
}
```

```swift
public struct PasteInjectionOutcome: Sendable, Equatable {
    public let result: PasteResult
    public let restoredPasteboard: Bool
}

public extension PasteInjector {
    static func pasteWithRestore(
        _ string: String,
        into pasteboard: NSPasteboard = .general,
        restoreDelay: Duration = .milliseconds(450)
    ) async -> PasteInjectionOutcome {
        let snapshot = PasteboardSnapshot.capture(from: pasteboard)
        let result = paste(string, into: pasteboard)
        try? await Task.sleep(for: restoreDelay)
        snapshot.restore(to: pasteboard)
        return PasteInjectionOutcome(result: result, restoredPasteboard: true)
    }
}

private struct PasteboardSnapshot {
    let items: [[(NSPasteboard.PasteboardType, Data)]]

    static func capture(from pasteboard: NSPasteboard) -> PasteboardSnapshot {
        let items = pasteboard.pasteboardItems?.map { item in
            item.types.compactMap { type in
                item.data(forType: type).map { (type, $0) }
            }
        } ?? []
        return PasteboardSnapshot(items: items)
    }

    func restore(to pasteboard: NSPasteboard) {
        pasteboard.clearContents()
        for itemTypes in items {
            let item = NSPasteboardItem()
            for (type, data) in itemTypes {
                item.setData(data, forType: type)
            }
            pasteboard.writeObjects([item])
        }
    }
}
```

```swift
import AppKit
import Core
import Foundation
import Observation
import OSLog

@MainActor
@Observable
final class VoiceCoordinator {
    enum State: Equatable {
        case idle
        case recording
        case transcribing
        case pasting
        case error(String)
    }

    private let audio = AudioCaptureService()
    private let engine: any VoiceTranscriptionEngine
    private let transcripts: VoiceTranscriptStore
    private let hud = MiniVoiceHUD()
    private let ptt = FnPTTMonitor()
    private let log = Logger(subsystem: "com.macallyouneed.voice", category: "coordinator")

    private(set) var state: State = .idle

    init(transcripts: VoiceTranscriptStore, engine: any VoiceTranscriptionEngine = Qwen3Engine()) {
        self.transcripts = transcripts
        self.engine = engine
        ptt.onDown = { [weak self] in Task { @MainActor in await self?.startRecording() } }
        ptt.onUp = { [weak self] in Task { @MainActor in await self?.stopRecordingAndPaste() } }
    }

    func start() {
        ptt.start()
    }

    func stop() {
        ptt.stop()
        hud.dismiss()
    }

    func startRecording() async {
        guard state == .idle else { return }
        guard await audio.requestPermission() else {
            fail("Microphone permission denied")
            return
        }
        do {
            try audio.start()
            state = .recording
            hud.show(.recording(level: 0))
        } catch {
            fail(error.localizedDescription)
        }
    }

    func stopRecordingAndPaste() async {
        guard state == .recording else { return }
        guard let captured = audio.stop(), captured.samples.count > 800 else {
            fail("No usable audio captured")
            return
        }

        state = .transcribing
        hud.show(.transcribing)
        do {
            let result = try await engine.transcribe(samples: captured.samples, sampleRate: captured.sampleRate)
            let text = result.text.trimmingCharacters(in: .whitespacesAndNewlines)
            guard !text.isEmpty else {
                fail("Transcript was empty")
                return
            }
            state = .pasting
            let pasteResult = await CursorPaster.paste(text)
            _ = try transcripts.save(VoiceTranscriptDraft(
                startedAt: captured.startedAt,
                endedAt: captured.endedAt,
                rawText: result.text,
                cleanedText: text,
                appBundleID: NSWorkspace.shared.frontmostApplication?.bundleIdentifier,
                language: result.language,
                modelIdentifier: result.modelIdentifier,
                audioPath: nil
            ))
            if pasteResult.didPostPasteEvent {
                hud.show(.pasted)
            } else {
                hud.show(.error("Text copied. Press Command-V to paste."))
            }
            try? await Task.sleep(for: .seconds(1))
            state = .idle
            hud.dismiss()
        } catch {
            fail(error.localizedDescription)
        }
    }

    private func fail(_ message: String) {
        log.error("Voice failed: \(message, privacy: .public)")
        state = .error(message)
        hud.show(.error(message))
        Task { @MainActor in
            try? await Task.sleep(for: .seconds(2))
            if case .error = state {
                state = .idle
                hud.dismiss()
            }
        }
    }
}
```
